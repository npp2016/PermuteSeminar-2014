Define a function that computes equation 8.20


foo <- function(b, x) {
    (dat$z[x] - b * dat$z[x - 1])^2
}

bvals <- sapply(pB, function(f) {

    # Get the estimates for the entire time series
    vals <- sapply(2:max(dat$period), foo, b = f)
